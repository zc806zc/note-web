box-shadow: 0 2px 40px 0 rgba(0, 0, 0, 0.1);

-o-transform: scale(1.1, 1.1);
-webkit-transform: scale(1.1, 1.1);
-moz-transform: translate(0px, -6px);
-o-transform: translate(0px, -6px);

}
